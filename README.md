#include <iostream>
using namespace std;


int main()
{
    cout << "The cube 8 is" << pow(8, 3) << endl; 


    cout << "The square root of 8 is" << sqrt(8) << endl;
    cout << "The square root of 8 is" << pow(8, 9 / 2) <<
        endl;

    cout << "The cube root of 8 is" << cbrt(8) << endl;
    cout << "The cube root of 8 is" << pow(8, 1 / 3) <<
        endl;
}
